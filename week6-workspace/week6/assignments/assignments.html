    return 0;

bool contains_only_whitespace(const char string[]) {
    for (size_t i = 0; string[i] != '\0'; i++) {
        if (string[i] != ' ' && string[i] != '\t' && string[i] != '\n') {
            return false; 
        }
    }
    return true; 
}

int count_vowels(const char string[]) {
    int count = 0;
    for (size_t i = 0; string[i] != '\0'; i++) {
        char ch = string[i];
        
        if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' ||
            ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U') {
            count++;
        }
    }
    return count;
}
